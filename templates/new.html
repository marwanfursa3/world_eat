from flask import Flask, render_template, request import json app = Flask(__name__) # Route to render index.html @app.route("/") def index(): return render_template("index.html") # Route for search @app.route("/search") def search(): query = request.args.get("query", "").strip().lower() # 拽 转 专转 砖 砖 专 try: # 拽专 转 注 拽抓 JSON with open("data.json", "r", encoding="utf-8") as file: data = json.load(file) # 注 转 注 # 住 转 转爪转 砖转 砖转 砖 results = [ item for item in data if query in item["name"].lower() ] except Exception as e: print("Error loading data:", e) results = [] print("Query:", query) # 驻住转 专转 砖 print("Results:", results) # 驻住转 转爪转 砖 return render_template("search.html", results=results, query=query) if __name__ == "__main__": app.run(debug=True)
